+++++ code-server/ci/build/build.sh
#!/bin/bash
set -euo pipefail

# This script builds code-server and outputs the built files to the dist
# directory. It is used by the CI to build code-server for release.

# The script is run from the root of the repository.

# The script assumes that the following environment variables are set:
# - CODE_SERVER_VERSION: the version of code-server to build
# - CODE_SERVER_COMMIT: the commit of code-server to build
# - CODE_SERVER_RELEASE: the release of code-server to build
# - CODE_SERVER_BUILD: the build number of code-server to build
# - CODE_SERVER_BUILD_DATE: the build date of code-server to build
# - CODE_SERVER_BUILD_TIME: the build time of code-server to build
# - CODE_SERVER_BUILD_TIMESTAMP: the build timestamp of code-server to build
# - CODE_SERVER_BUILD_PLATFORM: the platform to build for
# - CODE_SERVER_BUILD_ARCH: the architecture to build for
# - CODE_SERVER_BUILD_TARGET: the target to build for
# - CODE_SERVER_BUILD_OUTPUT: the output directory to build to

# The script also assumes that the following tools are installed:
# - git
# - node
# - yarn
# - make
# - gcc
# - g++
# - python
# - pkg-config
# - libx11-dev
# - libxkbfile-dev
# - libsecret-1-dev
# - libkrb5-dev

# The script will:
# 1. Clone the code-server repository
# 2. Checkout the commit to build
# 3. Install dependencies
# 4. Build code-server
# 5. Output the built files to the dist directory

# Clone the code-server repository
git clone https://github.com/coder/code-server.git
cd code-server

# Checkout the commit to build
git checkout "$CODE_SERVER_COMMIT"

# Install dependencies
yarn install --frozen-lockfile

# Build code-server
yarn build

# Output the built files to the dist directory
mkdir -p "$CODE_SERVER_BUILD_OUTPUT"
cp -r out/* "$CODE_SERVER_BUILD_OUTPUT"


+++++ code-server/ci/build/build.yaml
name: Build
on:
  push:
    branches:
      - main
  pull_request:
    branches:
      - main
jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v2
      - name: Set up Node.js
        uses: actions/setup-node@v2
        with:
          node-version: 16
      - name: Install dependencies
        run: yarn install --frozen-lockfile
      - name: Build
        run: yarn build
      - name: Upload build artifacts
        uses: actions/upload-artifact@v2
        with:
          name: code-server
          path: out/
